#!/usr/bin/python3

"""
Test cases for MyList class.
"""

MyList = __import__('1-my_list').MyList

# Test Case 1
my_list = MyList()
my_list.append(1)
my_list.append(4)
my_list.append(2)
my_list.append(3)
my_list.append(5)
assert my_list == [1, 4, 2, 3, 5]
my_list.print_sorted()
assert my_list == [1, 2, 3, 4, 5]

# Test Case 2
my_list2 = MyList()
my_list2.append(10)
my_list2.append(8)
my_list2.append(6)
my_list2.append(4)
my_list2.append(2)
assert my_list2 == [10, 8, 6, 4, 2]
my_list2.print_sorted()
assert my_list2 == [2, 4, 6, 8, 10]

# Additional test case
my_list3 = MyList()
my_list3.append(5)
my_list3.append(3)
my_list3.append(7)
assert my_list3 == [5, 3, 7]
my_list3.print_sorted()
assert my_list3 == [3, 5, 7]

print("All test cases pass")
